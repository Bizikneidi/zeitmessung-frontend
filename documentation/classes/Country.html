export class Country {

  public Name: string;
  public ISO: string;

  public constructor(name: string, iso: string) {
    if (name.length > 20) { // Shorten for better display
      this.Name = name.substr(0, 20) + '...';
    } else {
      this.Name = name;
    }
    this.ISO = iso;
  }
}
